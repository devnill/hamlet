## src/hamlet/world_state/grid.py

Add helper and two public methods to PositionGrid:

```python
def _footprint_positions(self, center: "Position", tier: int) -> list["Position"]:
    """Return all positions in (2*tier-1) x (2*tier-1) footprint centered on center."""
    half = tier - 1
    return [
        Position(center.x + dx, center.y + dy)
        for dx in range(-half, half + 1)
        for dy in range(-half, half + 1)
    ]

def occupy_footprint(self, position: "Position", tier: int, entity_id: str) -> None:
    """Mark all cells in the tier footprint as occupied by entity_id."""
    for pos in self._footprint_positions(position, tier):
        occupied_by = self._grid.get(pos)
        if occupied_by is not None and occupied_by != entity_id:
            raise ValueError(f"Position {pos} already occupied by '{occupied_by}'")
        self._grid[pos] = entity_id

def vacate_footprint(self, position: "Position", tier: int, entity_id: str) -> None:
    """Clear all footprint cells that are occupied by entity_id."""
    for pos in self._footprint_positions(position, tier):
        if self._grid.get(pos) == entity_id:
            del self._grid[pos]
```

async def upgrade_structure_tier(self, structure_id: str, new_tier: int) -> None:
    async with self._lock:
        structure = self._state.structures.get(structure_id)
        if structure is None:
            return
        old_tier = structure.size_tier
        # Vacate old footprint
        self._grid.vacate_footprint(structure.position, old_tier, structure_id)
        # Find agents in new footprint
        new_positions = self._grid._footprint_positions(structure.position, new_tier)
        agents_to_move = []
        for pos in new_positions:
            # check who occupies each cell
            occupant_id = self._grid.get_entity_at(pos)  # or equivalent method
            if occupant_id and occupant_id != structure_id and occupant_id in self._state.agents:
                agents_to_move.append(occupant_id)
        # Move displaced agents
        for agent_id in agents_to_move:
            agent = self._state.agents[agent_id]
            self._grid.vacate(agent.position)
            new_pos = self._find_free_position_outside(structure.position, new_tier)
            import dataclasses
            agent = dataclasses.replace(agent, position=new_pos)
            self._state.agents[agent_id] = agent
            self._grid.occupy(new_pos, agent_id)
        # Occupy new footprint
        self._grid.occupy_footprint(structure.position, new_tier, structure_id)
        # Update structure
        import dataclasses
        updated = dataclasses.replace(structure, size_tier=new_tier)
        self._state.structures[structure_id] = updated
    # Queue persistence outside lock
    await self._persistence.queue_write("structure", structure_id, self._state.structures[structure_id])
```

def _find_free_position_outside(self, center: "Position", tier: int) -> "Position":
    """Find nearest free cell outside the (2*tier-1)x(2*tier-1) footprint."""
    half = tier  # start just outside
    for radius in range(half, half + 30):
        for dx in range(-radius, radius + 1):
            for dy in range(-radius, radius + 1):
                if abs(dx) == radius or abs(dy) == radius:
                    pos = Position(center.x + dx, center.y + dy)
                    if not self._grid.is_occupied(pos):
                        return pos
    return Position(center.x + 50, center.y + 50)  # fallback
```
